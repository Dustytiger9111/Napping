    // ================= CONFIGURATION =================
    const COLORS = ["#FF6B6B", "#4ECDC4", "#45B7D1", "#FFA07A", "#98D8C8", "#F7DC6F", "#BB8FCE", "#85C1E9", "#F8B88B", "#80C997"];

    let placements = {};
    let currentProduct = null;
    let panelistData = {};

    let canvas, ctx, centerX, centerY, scale;

    // ================= QR CODE FUNCTIONS =================

    function generateQRCode(sessionID) {
        const qrContainer = document.getElementById("qrCode");
        qrContainer.innerHTML = "";

        const baseUrl = window.location.origin + window.location.pathname;
        const qrLink = `${baseUrl}?session=${sessionID}`;

        try {
            const qr = new QRCode(qrContainer, {
                text: qrLink,
                width: 200,
                height: 200,
                colorDark: "#667eea",
                colorLight: "#ffffff",
                correctLevel: QRCode.CorrectLevel.H
            });

            const now = new Date();
            document.getElementById("qrSessionInfo").innerHTML = 
                `<strong>Session:</strong> ${sessionID}<br>` +
                `<strong>Généré:</strong> ${now.toLocaleString()}<br>` +
                `<small>${qrLink}</small>`;

            showAlert("✅ QR Code généré avec succès !", "success");
        } catch (error) {
            showAlert("❌ Erreur lors de la génération du QR Code.", "error");
            console.error(error);
        }
    }

    function downloadQRCode() {
        const canvas = document.querySelector("#qrCode canvas");
        
        if (!canvas) {
            showAlert("❌ QR Code non généré.", "error");
            return;
        }

        const link = document.createElement("a");
        link.href = canvas.toDataURL("image/png");
        link.download = `qr_code_${currentSessionID}_${new Date().toISOString().split('T')[0]}.png`;
        document.body.appendChild(link);
        link.click();
        document.body.removeChild(link);
        
        showAlert("✅ QR Code téléchargé !", "success");
    }

    function copyQRLink() {
        const baseUrl = window.location.origin + window.location.pathname;
        const qrLink = `${baseUrl}?session=${currentSessionID}`;
        
        navigator.clipboard.writeText(qrLink).then(() => {
            showAlert("✅ Lien copié dans le presse-papiers !", "success");
        }).catch(() => {
            showAlert("❌ Erreur lors de la copie.", "error");
        });
    }

    function printQRCode() {
        const canvas = document.querySelector("#qrCode canvas");
        
        if (!canvas) {
            showAlert("❌ QR Code non généré.", "error");
            return;
        }

        const printWindow = window.open('', '', 'height=600,width=600');
        printWindow.document.write('<html><head><title>QR Code - ' + currentSessionID + '</title></head><body>');
        printWindow.document.write('<h2>QR Code Séance: ' + currentSessionID + '</h2>');
        printWindow.document.write('<img src="' + canvas.toDataURL() + '" style="max-width: 400px;">');

    // ================= CANVAS =================
    function initCanvas() {
        canvas = document.getElementById("mapCanvas");
        if (!canvas) return;

        canvas.width = CANVAS_SIZE;
        canvas.height = CANVAS_SIZE;
        
        ctx = canvas.getContext("2d");
        if (!ctx) return;

        centerX = canvas.width / 2;
        centerY = canvas.height / 2;
        scale = canvas.width / (2 * GRID_RANGE);

        canvas.addEventListener("click", handleCanvasClick);
        canvas.addEventListener("touchend", handleCanvasTouch);

        drawGrid();
    }

    function drawGrid() {
        if (!ctx || !canvas) return;
        
        ctx.fillStyle = "white";
        ctx.fillRect(0, 0, canvas.width, canvas.height);

        // Grille fine
        ctx.setLineDash([2, 2]);
        ctx.strokeStyle = "#e8e8e8";
        ctx.lineWidth = 0.5;

        for (let i = -GRID_RANGE; i <= GRID_RANGE; i += 5) {
            if (i % 10 !== 0) {
                const p = centerX + i * scale;
                ctx.beginPath();
                ctx.moveTo(p, 0);
                ctx.lineTo(p, canvas.height);
                ctx.stroke();

                ctx.beginPath();
                ctx.moveTo(0, centerY - i * scale);
                ctx.lineTo(canvas.width, centerY - i * scale);
                ctx.stroke();
            }
        }

        // Grille principale
        ctx.setLineDash([]);
        ctx.strokeStyle = "#667eea";
        ctx.lineWidth = 1.5;

        for (let i = -GRID_RANGE; i <= GRID_RANGE; i += 10) {
            const p = centerX + i * scale;
            ctx.beginPath();
            ctx.moveTo(p, 0);
            ctx.lineTo(p, canvas.height);
            ctx.stroke();

            ctx.beginPath();
            ctx.moveTo(0, centerY - i * scale);
            ctx.lineTo(canvas.width, centerY - i * scale);
            ctx.stroke();
        }

        // Axes
        ctx.strokeStyle = "#333";
        ctx.lineWidth = 2.5;

        ctx.beginPath();
        ctx.moveTo(centerX, 0);
        ctx.lineTo(centerX, canvas.height);
        ctx.stroke();

        ctx.beginPath();
        ctx.moveTo(0, centerY);
        ctx.lineTo(canvas.width, centerY);
        ctx.stroke();

        // Labels
        ctx.fillStyle = "#333";
        ctx.font = "bold 11px sans-serif";
        ctx.textAlign = "center";
        ctx.textBaseline = "top";

        for (let i = -GRID_RANGE; i <= GRID_RANGE; i += 10) {
            if (i !== 0) {
                const p = centerX + i * scale;
                ctx.fillText(i, p, 5);
            }
        }

        ctx.textAlign = "right";
        ctx.textBaseline = "middle";
        for (let i = -GRID_RANGE; i <= GRID_RANGE; i += 10) {
            if (i !== 0) {
                const p = centerY - i * scale;
                ctx.fillText(i, canvas.width - 5, p);
            }
        }

        redrawPlacements();
    }

    function redrawPlacements() {
        if (!ctx || !canvas) return;

        products.forEach((p, i) => {
            if (placements[i] !== undefined) {
                const x = centerX + placements[i].x * scale;
                const y = centerY - placements[i].y * scale;

                if (x >= 0 && x <= canvas.width && y >= 0 && y <= canvas.height) {
                    ctx.fillStyle = p.color;
                    ctx.beginPath();
                    ctx.arc(x, y, 10, 0, 2 * Math.PI);
                    ctx.fill();

                    ctx.strokeStyle = "white";
                    ctx.lineWidth = 2;
                    ctx.stroke();

                    ctx.fillStyle = "white";
                    ctx.font = "bold 10px sans-serif";
                    ctx.textAlign = "center";
                    ctx.textBaseline = "middle";
                    ctx.fillText(p.code, x, y);
                }
            }
        });
    }

    function handleCanvasClick(e) {
        if (currentProduct === null || !canvas) return;

        const rect = canvas.getBoundingClientRect();
        const xPix = e.clientX - rect.left;
        const yPix = e.clientY - rect.top;

        let x = Math.round((xPix - centerX) / scale);
        let y = Math.round((centerY - yPix) / scale);

        x = Math.max(-GRID_RANGE, Math.min(GRID_RANGE, x));
        y = Math.max(-GRID_RANGE, Math.min(GRID_RANGE, y));

        placements[currentProduct] = { x, y };

        drawGrid();
        updateProgress();
    }

    function handleCanvasTouch(e) {
        if (currentProduct === null || !canvas) return;
        e.preventDefault();

        const rect = canvas.getBoundingClientRect();
        const touch = e.changedTouches[0];
        const xPix = touch.clientX - rect.left;
        const yPix = touch.clientY - rect.top;

        let x = Math.round((xPix - centerX) / scale);
        let y = Math.round((centerY - yPix) / scale);

        x = Math.max(-GRID_RANGE, Math.min(GRID_RANGE, x));
        y = Math.max(-GRID_RANGE, Math.min(GRID_RANGE, y));

        placements[currentProduct] = { x, y };

        drawGrid();
        updateProgress();
    }

        if (currentProduct !== null) {
            products[currentProduct].verbatim = document.getElementById("verbatimInput").value;
        }

        currentProduct = i;

        document.querySelectorAll(".product-btn").forEach((btn, idx) => {

        updateProgress();
    }

    function updateProgress() {
        const total = products.length;
        const placed = Object.keys(placements).length;
        const percent = (placed / total) * 100;

        document.getElementById("progressFill").style.width = percent + "%";

        if (placed === total) {
            document.getElementById("continueBtn").style.display = "block";
        } else {
            document.getElementById("continueBtn").style.display = "none";
        }
    }

        try {
            const testResult = {
                panelistID: panelistData.id,
                date: panelistData.date,
                timestamp: panelistData.timestamp,
                products: products,
                placements: placements,
                comments: panelistData.comments

            showAlert("❌ Erreur lors de l'enregistrement.", "error");

            btn.disabled = false;
            btn.innerHTML = "✅ Soumettre";
        }
    }

        const container = document.getElementById("leaderSessionsList");

            item.className = "session-item";
            item.innerHTML = `
                <div class="session-item-info">

                </div>
                <div class="session-item-buttons">
                    <button class="btn-primary" onclick="loadSessionDetails('${sessionID}')">Voir</button>
                    <button class="btn-danger" onclick="deleteLeaderSession('${sessionID}')">Supprimer</button>

    function downloadSessionCSV() {
        if (!sessionData.tests.length) {
            showAlert("❌ Aucun test à exporter.", "error");
            return;
        }

        let rows = [["Session", currentSessionID], ["Date export", new Date().toLocaleString()], []];
        rows.push(["ID Panéliste", "Date", "Produit", "Position X", "Position Y", "Perceptions"]);

        sessionData.tests.forEach(test => {
            test.products.forEach((p, i) => {
                rows.push([
                    test.panelistID,
                    test.date,
                    p.code,
                    test.placements[i]?.x || 0,
                    test.placements[i]?.y || 0,
                    p.verbatim.replace(/[\r\n]+/g, " ")
                ]);
            });
        });

        const csv = rows.map(r => r.map(cell => `"${cell}"`).join(",")).join("\n");
        const blob = new Blob([csv], { type: "text/csv;charset=utf-8;" });
        const url = URL.createObjectURL(blob);
        const link = document.createElement("a");
        link.href = url;
        link.download = `napping_${currentSessionID}_${new Date().toISOString().split('T')[0]}.csv`;
        document.body.appendChild(link);
        link.click();
        document.body.removeChild(link);
    }

    function exportAllSessions() {
        const leaderSessions = JSON.parse(localStorage.getItem("leaderSessions") || "{}");
        const keys = Object.keys(leaderSessions);

        if (keys.length === 0) {
            showAlert("❌ Aucune séance à exporter.", "error");
            return;
        }

        const workbook = XLSX.utils.book_new();

        keys.forEach(sessionID => {
            const data = leaderSessions[sessionID];

            const rows = [["Session ID", sessionID], ["Créée le", new Date(data.createdAt).toLocaleString()], []];
            rows.push(["ID Panéliste", "Date", "Produit", "Position X", "Position Y", "Perceptions"]);

            data.tests.forEach(test => {
                test.products.forEach((p, i) => {
                    rows.push([
                        test.panelistID,
                        test.date,
                        p.code,
                        test.placements[i]?.x || 0,
                        test.placements[i]?.y || 0,
                        p.verbatim.replace(/[\r\n]+/g, " ")
                    ]);
                });
            });

            const worksheet = XLSX.utils.aoa_to_sheet(rows);
            XLSX.utils.book_append_sheet(workbook, worksheet, sessionID.substring(0, 31));
        });

        XLSX.writeFile(workbook, `napping_all_sessions_${new Date().toISOString().split('T')[0]}.xlsx`);
        showAlert("✅ Export Excel réussi.", "success");
    }
